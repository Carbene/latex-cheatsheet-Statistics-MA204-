\section {Point Estimation}
\[L(\theta) = \prod_{i=1}^n f(x_i; \theta)\]
\[\ell(\theta) = \ln L(\theta) = \sum_{i=1}^n \ln f(x_i; \theta)\]
\[h(\theta) = h(\hat{\theta}), \forall \text{MLE }\hat{\theta}\]
\[\frac{1}{n}\sum X_i^2 - \overline{X}^2 = \frac{1}{n}\sum_{i=1}^n (X_i - \overline{X})^2\]
\[\hat{\mu}_k = \text{E}(X^k)\]
\[p(\theta|x) = \frac{f(x|\theta) \pi(\theta)}{f(x)} \propto f(x|\theta) \times \pi(\theta)\]
\[\text{Unbiasedness: } \text{E}(\hat{\theta}) = \theta\]
\[\text{Bias}(\hat{\theta}) = \text{E}(\hat{\theta}) - \theta\]
\[\text{MSE}(\hat{\theta}) = \text{E}[(\hat{\theta} - \theta)^2]\]
\[= \text{Var}(\hat{\theta}) + [\text{Bias}(\hat{\theta})]^2\]
\[\text{Score Function:}S(\theta) = \frac{dl(\theta)}{d\theta} = \frac{L'(\theta)}{L(\theta)}\]
\[\text{Fisher Information: }I_n(\theta) = \text{Var}(S(\theta)) = nI(\theta)\]
\[ = \text{E}({\frac{L'(\theta)}{L(\theta)}}^2)(\text{E}(S(\theta)) = 0)(\mathbb{S} \perp \theta)\]
\[=-E\left[ \frac{d^2}{d \theta^2} \log f(X; \theta) \right]\]
\[\text{CRLB: }Var(\hat{\theta}) \ge \frac{{\tau'(\theta)}^2}{I_n(\theta)}(\tau'(\theta) = 1)\]
\[\text{Sufficiency: } f(x;\theta)\]
\[ = g(T(x),\theta)\times h(x), \forall \text{Statistics T}\]
\[\text{Eff}_{\hat{\theta}}(\theta) = \frac{\frac{1}{I_n(\theta)}}{Var(\hat{\theta})}\]
\[\text{UMVUE} \Leftarrow \text{Efficiency}\]
\[\hat{\theta}_n \xrightarrow{P} \theta, n \rightarrow \infty\]
\[\sqrt{nI(\theta)}(\hat{\theta}_n - \theta) \xrightarrow{d} N(0, 1)\]
\[\text{ or } \hat{\theta}_n \dot{\sim} N\left(\theta, \frac{1}{I_n(\theta)}\right)\]
