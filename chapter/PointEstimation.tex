\section {Point Estimation}
\[L(\theta) = \prod_{i=1}^n f(x_i; \theta)\]
\[\ell(\theta) = \ln L(\theta) = \sum_{i=1}^n \ln f(x_i; \theta)\]
\[h(\theta) = h(\hat{\theta}), \forall \text{MLE }\hat{\theta}\]
\[\frac{1}{n}\sum X_i^2 - \overline{X}^2 = \frac{1}{n}\sum_{i=1}^n (X_i - \overline{X})^2\]
\[\hat{\mu}_k = \text{E}(X^k)\]
\[p(\theta|x) = \frac{f(x|\theta) \pi(\theta)}{f(x)} \propto f(x|\theta) \times \pi(\theta)\]
\[\text{Unbiasedness: } \text{E}(\hat{\theta}) = \theta\text{ / } \text{Bias}(\hat{\theta}) = \text{E}(\hat{\theta}) - \theta\]
\[\text{MSE}(\hat{\theta}) = \text{E}[(\hat{\theta} - \theta)^2] = \text{Var}(\hat{\theta}) + [\text{Bias}(\hat{\theta})]^2\]
\[\text{Fisher Information: }I_n(\theta) = E\left[ - \frac{\partial^2 \log L(\theta)}{\partial \theta^2} \right]\]
\[\text{CRLB: }Var(\hat{\theta}) \ge \frac{1}{I_n(\theta)}\]
\[\text{Sufficiency: } f(x;\theta) = g(T(x),\theta)\times h(x), \forall \text{Statistics T}\]
\[\hat{\theta}_n \xrightarrow{P} \theta, n \xrightarrow{\infty}\]
\[\sqrt{nI(\theta)}(\hat{\theta}_n - \theta) \xrightarrow{d} N(0, 1) \text{ or } \hat{\theta}_n \dot{\sim} N\left(\theta, \frac{1}{I_n(\theta)}\right)\]
